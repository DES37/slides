display, animate and evaluation
sketch_7_2_Modularity_with_Functions.pde
void draw() {
background (255);
displayBall(); //call the display function
moveBall(); //call the move ball function
checkEdges(); // call the check edge function
}
void displayBall() { //display function defined that draws a ball very frame.
stroke(0);
fill(127);
ellipse(x, y, 32, 32);
}
void moveBall() { //move ball function defined that advances the circle's position.
x=x+xspeed;
y=y+yspeed;
}
void checkEdges() { //check edge function defined
if (x>width||x<0) {
xspeed=xspeed*-1;
}
if (y>height||y<0) {
yspeed=yspeed*-1;
}
}
